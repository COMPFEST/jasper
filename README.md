## Install

```bash
npm install --save jasper-ui
```

## Usage

```jsx
import React, { Component } from 'react'

import { Button } from 'jasper-ui'
import 'jasper-ui/dist/index.css'

class Example extends Component {
  render() {
    return (
      <Button>Yohohoho</Button>
    )
  }
}
```
